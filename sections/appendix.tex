\begin{frame}{Agent Dynamics}
	\begin{columns}
		\column{0.5\textwidth}
		\begin{align*}
			\dot{x_i} & = v_{x_i} \\
			\dot{y_i} & = v_{y_i} \\
			\dot{z_i} & = v_{z_i}
		\end{align*}

		\begin{align*}
			\sqrt{v_{x_i}^2 + v_{y_i}^2 + v_{z_i}^2} \le v_P, \forall i \in N
		\end{align*}

		\column{0.5\textwidth}
		\begin{align*}
			\dot{x_j} & = v_E \cos \phi_j \\
			\dot{y_j} & = v_E \cos \phi_j
		\end{align*}
	\end{columns}
\end{frame}

\begin{frame}{Agent Control Laws}
	\begin{columns}
		\column{0.5\textwidth}
		\begin{align*}
			v_{x_i}^* & = -v_P \frac{p_{x_i}}{\sqrt{p_{x_i}^2 + p_{y_i}^2 + p_{z_i}^2}} \\
			v_{y_i}^* & = -v_P \frac{p_{y_i}}{\sqrt{p_{x_i}^2 + p_{y_i}^2 + p_{z_i}^2}} \\
			v_{z_i}^* & = -v_P \frac{p_{z_i}}{\sqrt{p_{x_i}^2 + p_{y_i}^2 + p_{z_i}^2}}
		\end{align*}

		\column{0.5\textwidth}
		\begin{equation}
			\phi_j^* = \atantwo(p_{y_j}, p_{x_j})
		\end{equation}
	\end{columns}
\end{frame}

\begin{frame}{Radio Path Loss}
	\begin{align*}
		l(r_a, r_b) = 20 \log_{10}\left(\frac{4 \pi}{w} \enorm{r_a - r_b}\right)
	\end{align*}
	where $r_a, r_b \in \mathbb{R}^3$ are the Cartesian coordinates of agents
	$a, b$. $w$ is the wavelength of the radio signal.
\end{frame}

\begin{frame}{Simulation Parameters}
	\begin{table}
		\begin{tabular}{ccl}
			\hline
			Parameter  & Meaning                                    & Value                         \\
			\hline
			$l_{max}$  & Path loss threshold                        & 75.0 dB                       \\
			$v_E$      & Maximum evader velocity                    & 20.0 m/s                      \\
			$v_P$      & Maximum pursuer velocity                   & 30.0 m/s                      \\
			$\Delta t$ & Discrete time step size                    & 0.1 s                         \\
			$T$        & Terminal time                              & 60 s                          \\
			$w$        & Wavelength                                 & 0.32764203 m \footnotemark[1] \\
			$x$ range  & Random x coordinate range                  & $[0, 500]$ m                  \\
			$y$ range  & Random y coordinate range                  & $[0, 500]$ m                  \\
			$z$ range  & Random z coordinate range \footnotemark[2] & $[20, 100]$ m                 \\
			\hline
		\end{tabular}
	\end{table}
	\footnotetext[1]{Corresponds to 915MHz.}
	\footnotetext[2]{For pursuers only, evaders are affixed to $z = 0$.}
\end{frame}

\begin{frame}{Graph Definition}
	\begin{align*}
		V = \{v_i, \forall i \in N \cup M\}
	\end{align*}
	\begin{align*}
		\mathcal{E} = \{(v_i, v_k), \forall i, k \in N, i \ne k\} \cup \{(v_i, v_j), \forall i \in N, \forall j \in M\}
	\end{align*}
    Weighting function:
	\begin{align*}
		\sigma(x) = \frac{1}{1 + e^{-(l_{max} - x)}}
	\end{align*}
	\begin{align*}
		\omega((v_a, v_b)) = \sigma(l(r_a, r_b))
	\end{align*}
\end{frame}

\begin{frame}{Laplacian and Projection Matrices}
	\begin{columns}
		\column{0.5\textwidth}
		\begin{align*}
			L_{i, j} = \begin{cases}
				           -\omega((v_i, v_j)),                                  & i, j \in N, i \ne j \\
				           -\omega((v_i, v_j)),                                  & i \in N, j \in M    \\
				           \sum_{k \in M \cup N \setminus i} \omega((v_i, v_k)), & i \in N, i = j      \\
				           \sum_{k \in N} \omega((v_j, v_k)),                    & j \in M, i = j      \\
				           0,                                                    & i, j \in M
			           \end{cases}
		\end{align*}

		\column{0.5\textwidth}
		\begin{align*}
			P_{i, j} = \begin{cases}
				           1,  & 1 \le i \le j \\
				           -j, & i = j + 1     \\
				           0,  & i > j + 1
			           \end{cases}
		\end{align*}

	\end{columns}
\end{frame}
