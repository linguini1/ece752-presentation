\begin{frame}{Game structure}
    \begin{itemize}
        \item $n$ pursuers (UAVs)
        \item $m$ evaders (ground agents)
        \item Played in infinite 3D volume, ground agents on 2D plane at $z = 0$
        \item Perfect information
        \item Simple agent dynamics
        \item \textbf{Only pursuers act as routers}
        \item Agents are \textbf{connected} if they are within a connection radius of one
              another
    \end{itemize}
\end{frame}

\begin{frame}{Game Cost (Utilities)}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{itemize}
            \item $t \in [0, T]$: time horizon
            \item $u, v$ pursuer, evader control signals
            \item $x(t)$: state of the game at $t$
            \item $\mathcal{L}$: running cost
            \item $Q$: terminal cost
            \item $V$: value function
        \end{itemize}
        \column{0.5\textwidth}
        \begin{align*}
            J(x, u, v, t) = \int_{t}^T \mathcal{L}(x, u, v, \tau) d\tau + Q(x(T))
        \end{align*}
        \begin{align*}
            V(x, t) = \min_{u \in \mathcal{U}}\max_{v \in \mathcal{V}} J(x, u, v, t)
        \end{align*}
    \end{columns}
\end{frame}

\begin{frame}{Hamilton-Jacobi-Isaacs Equation}
    \begin{columns}
        \column{0.3\textwidth}
        \begin{itemize}
            \item HJI equation
            \item Solving this gives the value function
            \item Optimal controls can be derived from the value function
            \item Boundary condition: $Q$
        \end{itemize}
        \column{0.7\textwidth}
        \begin{align*}
            -\frac{\partial V}{\partial t} = \min_{u \in \mathcal{U}} \max_{v \in
                \mathcal{V}} \left\{\left(\frac{\partial V}{\partial x}\right)^\top \cdot
            f(x, u, v) + \mathcal{L}(x)\right\}
        \end{align*}
        \begin{align*}
            V(x, t) & = \min_{u \in \mathcal{U}}\max_{v \in \mathcal{V}} J(x, u, v, t) \\
            V(x, T) & = Q(x(T))
        \end{align*}
    \end{columns}
\end{frame}

\begin{frame}{Connectivity Measure}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{itemize}
            \item We need a direct measure of "connectivity" to optimize
            \item Graph Laplacian $L \in \mathbb{R}^{(n + m) \times (n + m)}$: matrix that
                  describes a graph
            \item $\lambda_2 = 0$: disconnected \cite[27]{graph-theory}
            \item $\lambda_2 > 0$: measure of connectivity \cite[33]{graph-theory}
        \end{itemize}
        \column{0.5\textwidth}

\end{frame}

\begin{frame}{Final Costs}
    \begin{columns}
        \column{0.4\textwidth}
        \begin{align*}
            \mathcal{L}(x, u, v) = 1
        \end{align*}
        Penalize slow convergence time.
        \column{0.4\textwidth}
        \begin{align*}
            Q(x(T)) = -\det(P^\top L(x(T)) P)
        \end{align*}
        Negated to respect that pursuers minimize cost
    \end{columns}
    \vspace{20pt}
    \center
    $P$ removes $\lambda_1$ \cite{pot-fields}, which is always 0 \cite[27]{graph-theory}.
    This is our closed-form cost.
\end{frame}
